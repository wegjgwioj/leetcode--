
迭代法反转单链表：
```python
def reverseList(head):
    prev = None
    curr = head
    while curr:
        next_node = curr.next  # 暂存后继节点
        curr.next = prev       # 反转指向
        prev = curr            # prev 后移
        curr = next_node       # curr 后移
    return prev
```

反转 区间链表
```python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

def reverseBetween(head, left, right):
    # 如果链表为空或不需要反转，直接返回
    if not head or left == right:
        return head
    
    # 1. 设置虚拟头节点
    dummy = ListNode(0)
    dummy.next = head
    pre = dummy
    
    # 2. pre 走到需要反转区间的前一个节点
    for _ in range(left - 1):
        pre = pre.next
        
    # 3. curr 指向反转区间的第一个节点
    curr = pre.next
    
    # 4. 穿针引线进行局部反转
    for _ in range(right - left):
        next_node = curr.next           # 记录要被提前的节点
        curr.next = next_node.next      # curr 往后跳一个节点
        next_node.next = pre.next       # 将 next_node 放到区间最前面
        pre.next = next_node            # pre 连接到刚被提前的 next_node
        
    # 5. 返回真正的头节点
    return dummy.next
```
