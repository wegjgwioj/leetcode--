'''
 n 总结点数 i 当前节点 计算左右子节点   物理还是数组
# 下沉函数heapify(arr,n,i)

##先判断然后在交换 递归当前最大值/最小


# 1 建堆 for i in range(n // 2 - 1, -1, -1): heapify(arr, n, i)

#2 排序：现在堆顶（下标 0）已经是最大的数了。把它和数组最后面的数交换。然后把数组长度减 1（不管最后那个最大的数了），对新的堆顶重新 heapify(arr, 剩余长度, 0)

# for i in range(n - 1, 0, -1): -> 交换 -> heapify(arr, i, 0)
'''

def heap_sort(arr):
	 n=len(arr)
	
 	# 建堆
	for i in range(n//2-1,-1,-1) : heapify(arr,n,i)

	#排序
	for i in range(n-1,0,-1):
		arr[0],arr[i] =arr[i],arr[0]
		heapify(arr,i,0)
	return arr

def heapify(arr,n,i):
	largest=i	
	left=2*i+1
	right=2*i+2
		
	if left<n and arr[left] >arr[largest]:
		largest = left
	if right<n and arr[right]>arr[largest]:
		largest = right
 	
	if largest != i:
		arr[i],arr[largest] = arr[largest],arr[i]
		heapify(arr,n,largest)
	
# top-k--最小堆

heap = arr[:k]


for i in range(k//2 -1,-1,-1):
